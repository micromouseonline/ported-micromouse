96  /// speed up the ADC to CLK/32 (4 times faster than normal)
97  ADCSRA |= (1 << ADPS2);

101  noInterrupts();
102  /// Timer 1 is used to generate stepper motor pulses
103  TCCR1A = 0x00; // normal port operation, no auto change of OCnA/B/C pins
104  TCNT1 = 0; // start the timer at 0x0000
105  OCR1A = MOTOR_IDLE_51Hz; // mutually prime to reduce interaction
106  OCR1B = MOTOR_IDLE_57Hz;
107  TCCR1B = 0x02; // 0: off, 1 => FCPU/1, 2 => FCPU/8, 3 => FCPU/64
108  TIMSK1 |= (1 << OCIE1A); // Timer 1 OC1A interrupt enable (0x10)
109  TIMSK1 |= (1 << OCIE1B); // Timer 1 OC1A interrupt enable (0x10)
