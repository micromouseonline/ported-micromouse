70 /***
71  * Sensor reading takes 8 states.
72  * The initial state of zero is only entered once and ensures all emitters off.
73  * Time needed per state is between 15 and 30 us and the update is called at
74  * a frequency of 4kHz so the load is between about 5% and 10%. Quite high.
75  * It looks like some channels are slower to read than others!
76  *
77  * Note that the ADC hardware is significantly speeded up in hardwareInit();
78  *
79  * With this scheme, sensors are read only 500 times per second. That is fine for a
80  * slow robot but significant improvements will be needed if speed increases.
81  *
82  * sensorUpdate takes about 600us to run and it could be made more efficient by
83  * use of a state machine or by only calling it when the answers are needed
84  *
85  * All told, reading the sensors takes about 25% of the available processor time.
86  *
87  * Also, time wasted here robs the processor of valuable power for maze flooding
88  * and control tasks.
89  */

91 void sensorUpdate() {
92  // gather all the unlit readings
93  int darkL = analogRead(LEFT_DIAG);
94  int darkR = analogRead(RIGHT_DIAG);
95  int darkFL = analogRead(LEFT_FRONT);
96  int darkFR = analogRead(RIGHT_FRONT);
97  digitalWrite(LED_TX_RF, 1); // front LEDs on
98  digitalWrite(LED_TX_LF, 1);
99  delayMicroseconds(50);
100  int litFL = analogRead(LEFT_FRONT);
101  int litFR = analogRead(RIGHT_FRONT);
102  digitalWrite(LED_TX_RF, 0); // front LEDs off
103  digitalWrite(LED_TX_LF, 0);
104  digitalWrite(LED_TX_RD, 1); // side LEDs on
105  digitalWrite(LED_TX_LD, 1);
106  delayMicroseconds(50);
107  int litL = analogRead(LEFT_DIAG);
108  int litR = analogRead(RIGHT_DIAG);
109  digitalWrite(LED_TX_RD, 0); // side LEDs off
110  digitalWrite(LED_TX_LD, 0);
111 
112  // all the dark and lit values have now been collected so
113  // see what we got from reflection but never accept negative readings
114  rawFL = max(litFL - darkFL, 0);
115  rawFR = max(litFR - darkFR, 0);
116  rawL = max(litL - darkL, 0);
117  rawR = max(litR - darkR, 0);
118 
119  // now calculate the normalised readings
120  sensFL = (rawFL * LF_NOMINAL) / LF_CAL;
121  sensFR = (rawFR * RF_NOMINAL) / RF_CAL;
122  sensL = (rawL * LD_NOMINAL) / LD_CAL;
123  sensR = (rawR * RD_NOMINAL) / RD_CAL;
124 
125  frontSum = sensFL + sensFR;
126  frontDiff = sensFL - sensFR;
127  // there is some hysteresis built in to the side sensors to ensure
128  // cleaner edges
129  // decide whether walls are present - use both sensors at the front
130 
131  wallSensorFrontRight = sensFR > FRONT_THRESHOLD;
132  wallSensorFrontLeft = sensFL > FRONT_THRESHOLD;
133  wallSensorFront = wallSensorFrontRight && wallSensorFrontLeft;
134 
135  if (sensL < (DIAG_THRESHOLD)) {
136  wallSensorLeft = false;
137  } else if (sensL > (DIAG_THRESHOLD + 5)) {
138  wallSensorLeft = true;
139  }
140  if (sensR < (DIAG_THRESHOLD)) {
141  wallSensorRight = false;
142  } else if (sensR > (DIAG_THRESHOLD + 5)) {
143  wallSensorRight = true;
144  }
145 
146 }
